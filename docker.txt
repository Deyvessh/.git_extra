// docker commands

-- docker images
locate images locally

-- docker search jenkins
search jenkins from hub.docker.com

-- docker pull ubuntu
pull ubuntu from hub.docker.com

-- docker run -it --name container_name image_name /bin/bash
create a container, run it and takes you inside the container

-- service docker status
status of docker services

-- service start docker
start the docker services

-- service stop docker
stop the docker services

-- service start container_name
start the docker_container

-- service attach container_name
attach the started docker_container after command - service start docker_container

-- service stop container_name
stop the docker_Container

-- docker ps -a
to see all stopped and started docker_container

-- docker ps
to see all running container - ps means process statust

-- docker rm docker_container
to delete a container

-- which docker
to check docker is installed or not

-- apt-get install docker.io -y
install docker

-- docker --version
get the version of docker

-- docker info
information of docker

-- docker image rm -f image_id
delete the image

-- docker rm container_name
removing docker
